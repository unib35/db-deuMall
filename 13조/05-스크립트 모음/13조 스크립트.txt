DROP TABLE ACCOUNT cascade constraints;
DROP TABLE BRANCHSTORAGE cascade constraints;
DROP TABLE BUSINESSSTORAGE cascade constraints;
DROP TABLE PRODUCTINFO cascade constraints;
DROP TABLE STORAGE cascade constraints;
DROP TABLE WAREHOUSING cascade constraints;

CREATE TABLE ACCOUNT
(	
	OFFICENUMBER VARCHAR2(50), 
	ID VARCHAR2(50) NOT NULL, 
	PW VARCHAR2(50) NOT NULL,
	
	CONSTRAINT ACCOUNT_PK PRIMARY KEY (OFFICENUMBER)
);

CREATE TABLE BRANCHSTORAGE
(	
	PID VARCHAR2(50) ,
	OFFICENUMBER VARCHAR2(50) ,  
	PAMOUNT VARCHAR2(50),
	
	CONSTRAINT BRANCHSTORAGE_PK PRIMARY KEY (PID,OFFICENUMBER),
	CONSTRAINT BRANCHSTORAGE_FK FOREIGN KEY (OFFICENUMBER) REFERENCES ACCOUNT(OFFICENUMBER) ON DELETE CASCADE,
	CONSTRAINT BRANCHSTORAGE_FK2 FOREIGN KEY (PID) REFERENCES PRODUCTINFO(PID)
);

CREATE TABLE BUSINESSSTORAGE 
(	
	PID VARCHAR2(50), 
	PAMOUNT VARCHAR2(50),

	CONSTRAINT BUSINESSSTORAGE_PK PRIMARY KEY (PID),
	CONSTRAINT BUSINESSSTORAGE_FK FOREIGN KEY (PID) REFERENCES PRODUCTINFO(PID)
);

CREATE TABLE PRODUCTINFO
(	
	PID VARCHAR2(50), 
	PNAME VARCHAR2(50), 
	PPRICE VARCHAR2(50),

	CONSTRAINT PRODUCTINFO_PK PRIMARY KEY (PID)
	
);


CREATE TABLE STORAGE
(	
	PID VARCHAR2(50), 
	PAMOUNT VARCHAR2(50),

	CONSTRAINT STORAGE_PK PRIMARY KEY (PID),
	CONSTRAINT STORAGE_FK FOREIGN KEY (PID) REFERENCES PRODUCTINFO(PID)
   
);

CREATE TABLE WAREHOUSING
(
	ORDERNUMBER VARCHAR2(50), 
	TYPE VARCHAR2(50), 
	OFFICENUMBER VARCHAR2(50), 
	PID VARCHAR2(50), 
	ORDERAMOUNT VARCHAR2(50), 
	PDATE VARCHAR2(50),
	
	CONSTRAINT WAREHOUSING_PK PRIMARY KEY (ORDERNUMBER),
	CONSTRAINT WAREHOUSING_FK FOREIGN KEY (PID) REFERENCES PRODUCTINFO(PID),
	CONSTRAINT WAREHOUSING_FK2 FOREIGN KEY (OFFCENUMBER) REFERENCES ACCOUNT(OFFICENUMBER) ON DELETE CASCADE

);
